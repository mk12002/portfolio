```python
from fastapi import FastAPI, Depends, HTTPException, Request
from fastapi.security import HTTPBearer
from pydantic import BaseModel, Field, validator
from typing import List
import logging
import uuid
from datetime import datetime

from app.auth import verify_token, get_current_user
from app.rate_limit import rate_limiter
from app.model import MLModel
from app.logging_config import security_logger

app = FastAPI(
    title="Secure ML Inference API",
    description="Production-ready ML API with security controls",
    version="1.0.0"
)

# Initialize model
ml_model = MLModel()

# Request/Response schemas with validation
class PredictionRequest(BaseModel):
    features: List[float] = Field(..., min_items=1, max_items=100)
    
    @validator('features', each_item=True)
    def validate_feature_range(cls, v):
        if not -1000 <= v <= 1000:
            raise ValueError('Feature value out of acceptable range')
        return v

class PredictionResponse(BaseModel):
    request_id: str
    prediction: str
    # Note: We intentionally DON'T return confidence scores
    # to prevent model extraction attacks

# Security middleware
@app.middleware("http")
async def security_headers(request: Request, call_next):
    response = await call_next(request)
    response.headers["X-Content-Type-Options"] = "nosniff"
    response.headers["X-Frame-Options"] = "DENY"
    response.headers["Cache-Control"] = "no-store"
    return response

@app.post("/predict", response_model=PredictionResponse)
@rate_limiter(max_requests=100, window_seconds=60)
async def predict(
    request: PredictionRequest,
    current_user: dict = Depends(get_current_user)
):
    request_id = str(uuid.uuid4())
    
    # Log the inference request (no PII, no raw features)
    security_logger.info({
        "event": "inference_request",
        "request_id": request_id,
        "user_id": current_user["user_id"],
        "feature_count": len(request.features),
        "timestamp": datetime.utcnow().isoformat()
    })
    
    try:
        # Run inference
        prediction = ml_model.predict(request.features)
        
        # Log success
        security_logger.info({
            "event": "inference_success",
            "request_id": request_id,
            "user_id": current_user["user_id"]
        })
        
        return PredictionResponse(
            request_id=request_id,
            prediction=prediction
        )
        
    except Exception as e:
        # Log failure without leaking details
        security_logger.error({
            "event": "inference_error",
            "request_id": request_id,
            "user_id": current_user["user_id"],
            "error_type": type(e).__name__
        })
        raise HTTPException(
            status_code=500,
            detail="Inference failed"  # Generic message
        )

@app.get("/health")
async def health():
    return {"status": "healthy"}
```

`app/auth.py`:

```python
from fastapi import HTTPException, Security
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
from jose import jwt, JWTError
from datetime import datetime, timedelta
import os

security = HTTPBearer()

SECRET_KEY = os.environ.get("JWT_SECRET", "change-me-in-production")
ALGORITHM = "HS256"

def create_token(user_id: str, role: str) -> str:
    payload = {
        "sub": user_id,
        "role": role,
        "exp": datetime.utcnow() + timedelta(hours=1),
        "iat": datetime.utcnow()
    }
    return jwt.encode(payload, SECRET_KEY, algorithm=ALGORITHM)

def verify_token(credentials: HTTPAuthorizationCredentials = Security(security)):
    token = credentials.credentials
    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        return payload
    except JWTError:
        raise HTTPException(status_code=401, detail="Invalid token")

async def get_current_user(credentials: HTTPAuthorizationCredentials = Security(security)):
    payload = verify_token(credentials)
    return {
        "user_id": payload["sub"],
        "role": payload["role"]
    }
```

`app/rate_limit.py`:
```python
from functools import wraps
from fastapi import HTTPException, Request
from collections import defaultdict
import time

# Simple in-memory rate limiter (use Redis in production)
request_counts = defaultdict(list)

def rate_limiter(max_requests: int, window_seconds: int):
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            # Get request from kwargs
            request = kwargs.get('request') or args[0] if args else None
            
            # Use IP + user as key (in production, prefer user_id)
            client_ip = request.client.host if request else "unknown"
            current_user = kwargs.get('current_user', {})
            user_id = current_user.get('user_id', client_ip)
            
            now = time.time()
            key = f"{user_id}"
            
            # Clean old requests
            request_counts[key] = [
                ts for ts in request_counts[key] 
                if now - ts < window_seconds
            ]
            
            if len(request_counts[key]) >= max_requests:
                raise HTTPException(
                    status_code=429,
                    detail="Rate limit exceeded"
                )
            
            request_counts[key].append(now)
            return await func(*args, **kwargs)
        return wrapper
    return decorator
```

```markdown
# Threat Model: Secure ML Inference API

## System Description
REST API serving ML model predictions with authentication and rate limiting.

## Assets
- ML model weights (intellectual property)
- User credentials and tokens
- Inference logs (may contain sensitive patterns)
- API keys

## Trust Boundaries
```
Internet ──┬── API Gateway ──┬── Application ──┬── Model
           │                 │                 │
         Untrusted      Semi-trusted       Trusted
```

## Threats (STRIDE + ML-specific)

### Spoofing
- **Threat:** Attacker uses stolen API keys
- **Mitigation:** Short-lived JWTs, key rotation, anomaly detection

### Tampering
- **Threat:** Attacker modifies requests in transit
- **Mitigation:** TLS required, request signing

### Repudiation
- **Threat:** User denies making requests
- **Mitigation:** Comprehensive logging with request IDs

### Information Disclosure
- **Threat:** Model extraction via API queries
- **Mitigation:** Rate limiting, output minimization (no confidence scores)

### Denial of Service
- **Threat:** Overwhelm API with requests
- **Mitigation:** Rate limiting, input size limits

### Elevation of Privilege
- **Threat:** User accesses admin endpoints
- **Mitigation:** RBAC, JWT role claims

### ML-Specific: Adversarial Inputs
- **Threat:** Crafted inputs cause misclassification
- **Mitigation:** Input validation, anomaly detection

### ML-Specific: Model Extraction
- **Threat:** Reconstruct model via many queries
- **Mitigation:** Query limits, output rounding/noise

## Detection
- Alert on: >100 requests/minute from single user
- Alert on: Unusual input patterns (potential adversarial)
- Alert on: Repeated auth failures
```

`log_parser.py`:
```python
import re
import json
from datetime import datetime
from typing import Dict, List, Optional
from dataclasses import dataclass

@dataclass
class LogEvent:
    timestamp: datetime
    source: str
    event_type: str
    user: Optional[str]
    ip: Optional[str]
    action: str
    outcome: str
    raw: str

class LogParser:
    """Parse multiple log formats into normalized events."""
    
    # Auth log pattern (Linux)
    AUTH_PATTERN = re.compile(
        r'(\w+\s+\d+\s+\d+:\d+:\d+)\s+\S+\s+\S+:\s+'
        r'(Failed|Accepted)\s+password\s+for\s+(\S+)\s+from\s+([\d.]+)'
    )
    
    # Web access log pattern (Apache/Nginx)
    WEB_PATTERN = re.compile(
        r'([\d.]+)\s+-\s+(\S+)\s+\[([^\]]+)\]\s+"(\w+)\s+([^"]+)"\s+(\d+)\s+(\d+)'
    )
    
    # JSON log pattern (modern apps)
    def parse_json_log(self, line: str) -> Optional[LogEvent]:
        try:
            data = json.loads(line)
            return LogEvent(
                timestamp=datetime.fromisoformat(data.get('timestamp', '')),
                source='json',
                event_type=data.get('event_type', 'unknown'),
                user=data.get('user'),
                ip=data.get('ip'),
                action=data.get('action', ''),
                outcome=data.get('outcome', ''),
                raw=line
            )
        except (json.JSONDecodeError, ValueError):
            return None
    
    def parse_auth_log(self, line: str) -> Optional[LogEvent]:
        match = self.AUTH_PATTERN.search(line)
        if match:
            return LogEvent(
                timestamp=self._parse_syslog_time(match.group(1)),
                source='auth',
                event_type='authentication',
                user=match.group(3),
                ip=match.group(4),
                action='login',
                outcome='success' if match.group(2) == 'Accepted' else 'failure',
                raw=line
            )
        return None
    
    def parse_web_log(self, line: str) -> Optional[LogEvent]:
        match = self.WEB_PATTERN.search(line)
        if match:
            status = int(match.group(6))
            return LogEvent(
                timestamp=self._parse_clf_time(match.group(3)),
                source='web',
                event_type='http_request',
                user=match.group(2) if match.group(2) != '-' else None,
                ip=match.group(1),
                action=f"{match.group(4)} {match.group(5)}",
                outcome='success' if 200 <= status < 400 else 'error',
                raw=line
            )
        return None
    
    def parse(self, line: str, source_hint: str = 'auto') -> Optional[LogEvent]:
        """Auto-detect and parse log line."""
        if source_hint == 'json' or line.startswith('{'):
            return self.parse_json_log(line)
        elif source_hint == 'auth' or 'password' in line.lower():
            return self.parse_auth_log(line)
        elif source_hint == 'web' or re.match(r'[\d.]+\s+-', line):
            return self.parse_web_log(line)
        return None
    
    def _parse_syslog_time(self, time_str: str) -> datetime:
        # Simplified - add year
        return datetime.strptime(f"2026 {time_str}", "%Y %b %d %H:%M:%S")
    
    def _parse_clf_time(self, time_str: str) -> datetime:
        return datetime.strptime(time_str, "%d/%b/%Y:%H:%M:%S %z")
```

**Step 2: Feature Extraction**

`feature_extraction.py`:
```python
import numpy as np
from collections import defaultdict
from typing import List, Dict
from datetime import datetime, timedelta
from log_parser import LogEvent

class FeatureExtractor:
    """Extract ML-ready features from log events."""
    
    def __init__(self, window_minutes: int = 5):
        self.window = timedelta(minutes=window_minutes)
        self.user_history = defaultdict(list)
        self.ip_history = defaultdict(list)
    
    def extract_features(self, events: List[LogEvent]) -> np.ndarray:
        """Extract feature vectors for anomaly detection."""
        features = []
        
        for event in events:
            feature_vector = self._extract_event_features(event)
            features.append(feature_vector)
            
            # Update history
            if event.user:
                self.user_history[event.user].append(event)
            if event.ip:
                self.ip_history[event.ip].append(event)
        
        return np.array(features)
    
    def _extract_event_features(self, event: LogEvent) -> List[float]:
        """Extract features for a single event."""
        features = []
        
        # Time-based features
        features.append(event.timestamp.hour)  # Hour of day
        features.append(event.timestamp.weekday())  # Day of week
        
        # User behavior features
        if event.user:
            user_events = self.user_history.get(event.user, [])
            recent = [e for e in user_events 
                     if event.timestamp - e.timestamp < self.window]
            
            features.append(len(recent))  # Events in window
            features.append(self._failure_rate(recent))  # Failure rate
            features.append(self._unique_ips(recent))  # Unique IPs
        else:
            features.extend([0, 0, 0])
        
        # IP behavior features
        if event.ip:
            ip_events = self.ip_history.get(event.ip, [])
            recent = [e for e in ip_events 
                     if event.timestamp - e.timestamp < self.window]
            
            features.append(len(recent))  # Events from IP
            features.append(self._unique_users(recent))  # Unique users
            features.append(self._failure_rate(recent))  # Failure rate
        else:
            features.extend([0, 0, 0])
        
        # Event-specific features
        features.append(1 if event.outcome == 'failure' else 0)
        features.append(self._encode_event_type(event.event_type))
        
        return features
    
    def _failure_rate(self, events: List[LogEvent]) -> float:
        if not events:
            return 0.0
        failures = sum(1 for e in events if e.outcome == 'failure')
        return failures / len(events)
    
    def _unique_ips(self, events: List[LogEvent]) -> int:
        return len(set(e.ip for e in events if e.ip))
    
    def _unique_users(self, events: List[LogEvent]) -> int:
        return len(set(e.user for e in events if e.user))
    
    def _encode_event_type(self, event_type: str) -> int:
        mapping = {
            'authentication': 1,
            'http_request': 2,
            'authorization': 3,
            'admin_action': 4
        }
        return mapping.get(event_type, 0)
```

```python
        scaled = self.scaler.fit_transform(features)
        
        # Train model
        self.model.fit(scaled)
        self.is_fitted = True
        
        logger.info("Training complete")

        """
        Predict anomalies.
        Returns: (labels, scores)
          labels: 1 = normal, -1 = anomaly
          scores: anomaly scores (lower = more anomalous)
        """
        if not self.is_fitted:
            raise ValueError("Model not trained. Call fit() first.")
        
        scaled = self.scaler.transform(features)
        labels = self.model.predict(scaled)
        scores = self.model.score_samples(scaled)
        
        return labels, scores

                         threshold: float = -0.5) -> List[int]:
        """Return indices of anomalous events."""
        labels, scores = self.predict(features)
        
        anomalies = []
        for i, (label, score) in enumerate(zip(labels, scores)):
            if label == -1 or score < threshold:
                anomalies.append(i)
        
        return anomalies
```

```python
import numpy as np
import torch
import torch.nn as nn
from typing import Tuple, Optional, Callable
from dataclasses import dataclass

@dataclass
class AttackResult:
    original_input: np.ndarray
    adversarial_input: np.ndarray
    original_pred: int
    adversarial_pred: int
    perturbation_norm: float
    success: bool

class AdversarialAttacks:
    """Implementation of common adversarial attacks."""
    
    def __init__(self, model: nn.Module, device: str = 'cpu'):
        self.model = model.to(device)
        self.device = device
        self.model.eval()
    
    def fgsm_attack(self, x: torch.Tensor, y: torch.Tensor, 
                    epsilon: float = 0.03) -> AttackResult:
        """
        Fast Gradient Sign Method attack.
        
        Args:
            x: Input tensor (batch, channels, height, width)
            y: True labels
            epsilon: Perturbation magnitude
        """
        x = x.clone().detach().to(self.device)
        x.requires_grad = True
        
        # Forward pass
        output = self.model(x)
        loss = nn.CrossEntropyLoss()(output, y)
        
        # Backward pass
        self.model.zero_grad()
        loss.backward()
        
        # Create adversarial example
        perturbation = epsilon * x.grad.sign()
        x_adv = x + perturbation
        x_adv = torch.clamp(x_adv, 0, 1)  # Keep in valid range
        
        # Evaluate
        with torch.no_grad():
            orig_pred = self.model(x).argmax(dim=1)
            adv_pred = self.model(x_adv).argmax(dim=1)
        
        return AttackResult(
            original_input=x.detach().cpu().numpy(),
            adversarial_input=x_adv.detach().cpu().numpy(),
            original_pred=orig_pred.item(),
            adversarial_pred=adv_pred.item(),
            perturbation_norm=float(torch.norm(perturbation).item()),
            success=orig_pred.item() != adv_pred.item()
        )
    
    def pgd_attack(self, x: torch.Tensor, y: torch.Tensor,
                   epsilon: float = 0.03, alpha: float = 0.01,
                   num_steps: int = 40) -> AttackResult:
        """
        Projected Gradient Descent attack (stronger than FGSM).
        
        Args:
            epsilon: Maximum perturbation
            alpha: Step size
            num_steps: Number of iterations
        """
        x = x.clone().detach().to(self.device)
        x_adv = x.clone()
        
        for _ in range(num_steps):
            x_adv.requires_grad = True
            output = self.model(x_adv)
            loss = nn.CrossEntropyLoss()(output, y)
            
            self.model.zero_grad()
            loss.backward()
            
            # Update with gradient step
            x_adv = x_adv + alpha * x_adv.grad.sign()
            
            # Project back to epsilon-ball
            perturbation = torch.clamp(x_adv - x, -epsilon, epsilon)
            x_adv = torch.clamp(x + perturbation, 0, 1).detach()
        
        with torch.no_grad():
            orig_pred = self.model(x).argmax(dim=1)
            adv_pred = self.model(x_adv).argmax(dim=1)
        
        return AttackResult(
            original_input=x.detach().cpu().numpy(),
            adversarial_input=x_adv.detach().cpu().numpy(),
            original_pred=orig_pred.item(),
            adversarial_pred=adv_pred.item(),
            perturbation_norm=float(torch.norm(x_adv - x).item()),
            success=orig_pred.item() != adv_pred.item()
        )


class AdversarialDefenses:
    """Implementation of adversarial defenses."""
    
    @staticmethod
    def jpeg_compression(x: np.ndarray, quality: int = 75) -> np.ndarray:
        """Defense via JPEG compression."""
        from PIL import Image
        import io
        
        # Convert to image
        img = Image.fromarray((x * 255).astype(np.uint8))
        
        # Compress and decompress
        buffer = io.BytesIO()
        img.save(buffer, format='JPEG', quality=quality)
        buffer.seek(0)
        img_compressed = Image.open(buffer)
        
        return np.array(img_compressed) / 255.0
    
    @staticmethod
    def bit_depth_reduction(x: np.ndarray, bits: int = 4) -> np.ndarray:
        """Reduce bit depth to remove small perturbations."""
        max_val = 2 ** bits - 1
        return np.round(x * max_val) / max_val
    
    @staticmethod
    def input_smoothing(x: np.ndarray, kernel_size: int = 3) -> np.ndarray:
        """Gaussian smoothing to remove perturbations."""
        from scipy.ndimage import gaussian_filter
        return gaussian_filter(x, sigma=kernel_size/3)


class RobustnessEvaluator:
    """Evaluate model robustness against attacks."""
    
    def __init__(self, model: nn.Module, attacks: AdversarialAttacks):
        self.model = model
        self.attacks = attacks
    
    def evaluate_robustness(self, test_loader, 
                           epsilons: list = [0.01, 0.03, 0.05, 0.1]):
        """Evaluate model robustness at different perturbation levels."""
        results = {}
        
        for eps in epsilons:
            total, successful = 0, 0
            
            for x, y in test_loader:
                result = self.attacks.fgsm_attack(x, y, epsilon=eps)
                total += 1
                if result.success:
                    successful += 1
            
            results[eps] = {
                'attack_success_rate': successful / total,
                'model_accuracy': 1 - (successful / total)
            }
        
        return results
    
    def generate_report(self, results: dict) -> str:
        report = "# Adversarial Robustness Report\n\n"
        report += "| Epsilon | Attack Success | Model Accuracy |\n"
        report += "|---------|----------------|----------------|\n"
        
        for eps, metrics in results.items():
            report += f"| {eps:.3f} | {metrics['attack_success_rate']:.1%} | "
            report += f"{metrics['model_accuracy']:.1%} |\n"
        
        return report
```

```python
import numpy as np
from collections import defaultdict
from datetime import datetime, timedelta
from typing import List, Dict, Optional
from dataclasses import dataclass
from sklearn.cluster import DBSCAN
import hashlib

@dataclass
class QueryEvent:
    user_id: str
    timestamp: datetime
    input_vector: np.ndarray
    input_hash: str

@dataclass
class ThreatAlert:
    user_id: str
    alert_type: str
    confidence: float
    evidence: str
    recommended_action: str

class ModelTheftDetector:
    """Detect potential model extraction attacks."""
    
    def __init__(self, 
                 query_rate_threshold: int = 1000,
                 time_window_hours: int = 1,
                 grid_detection_threshold: float = 0.8):
        self.query_rate_threshold = query_rate_threshold
        self.time_window = timedelta(hours=time_window_hours)
        self.grid_threshold = grid_detection_threshold
        
        self.user_queries: Dict[str, List[QueryEvent]] = defaultdict(list)
        self.alerts: List[ThreatAlert] = []
    
    def log_query(self, user_id: str, input_vector: np.ndarray) -> Optional[ThreatAlert]:
        """Log a query and check for suspicious patterns."""
        
        # Create query event
        event = QueryEvent(
            user_id=user_id,
            timestamp=datetime.utcnow(),
            input_vector=input_vector,
            input_hash=self._hash_input(input_vector)
        )
        
        self.user_queries[user_id].append(event)
        
        # Run detection checks
        alerts = []
        alerts.append(self._check_query_rate(user_id))
        alerts.append(self._check_grid_search(user_id))
        alerts.append(self._check_boundary_probing(user_id))
        
        # Return highest confidence alert
        valid_alerts = [a for a in alerts if a is not None]
        if valid_alerts:
            return max(valid_alerts, key=lambda x: x.confidence)
        return None
    
    def _check_query_rate(self, user_id: str) -> Optional[ThreatAlert]:
        """Detect abnormally high query rates."""
        queries = self.user_queries[user_id]
        cutoff = datetime.utcnow() - self.time_window
        recent = [q for q in queries if q.timestamp > cutoff]
        
        if len(recent) > self.query_rate_threshold:
            return ThreatAlert(
                user_id=user_id,
                alert_type="HIGH_QUERY_RATE",
                confidence=min(1.0, len(recent) / (self.query_rate_threshold * 2)),
                evidence=f"{len(recent)} queries in {self.time_window}",
                recommended_action="Apply rate limiting, require CAPTCHA"
            )
        return None
    
    def _check_grid_search(self, user_id: str) -> Optional[ThreatAlert]:
        """Detect systematic grid search patterns."""
        queries = self.user_queries[user_id]
        if len(queries) < 100:
            return None
        
        # Get recent inputs
        recent_inputs = np.array([q.input_vector for q in queries[-500:]])
        
        # Check for grid-like patterns
        # High regularity = potential grid search
        regularity_score = self._calculate_grid_regularity(recent_inputs)
        
        if regularity_score > self.grid_threshold:
            return ThreatAlert(
                user_id=user_id,
                alert_type="GRID_SEARCH_PATTERN",
                confidence=regularity_score,
                evidence=f"Regularity score: {regularity_score:.2f}",
                recommended_action="Flag account, add output noise"
            )
        return None
    
    def _check_boundary_probing(self, user_id: str) -> Optional[ThreatAlert]:
        """Detect boundary probing (model extraction technique)."""
        queries = self.user_queries[user_id]
        if len(queries) < 50:
            return None
        
        recent_inputs = np.array([q.input_vector for q in queries[-200:]])
        
        # Check for pairs of very similar inputs (boundary probing)
        # Model thieves often query similar inputs to find decision boundaries
        close_pairs = 0
        total_pairs = 0
        
        for i in range(len(recent_inputs) - 1):
            for j in range(i + 1, min(i + 10, len(recent_inputs))):
                dist = np.linalg.norm(recent_inputs[i] - recent_inputs[j])
                if dist < 0.01:  # Very close inputs
                    close_pairs += 1
                total_pairs += 1
        
        if total_pairs > 0:
            boundary_score = close_pairs / total_pairs
            if boundary_score > 0.3:
                return ThreatAlert(
                    user_id=user_id,
                    alert_type="BOUNDARY_PROBING",
                    confidence=min(1.0, boundary_score * 2),
                    evidence=f"{close_pairs} very similar query pairs",
                    recommended_action="Apply output rounding, monitor closely"
                )
        return None
    
    def _calculate_grid_regularity(self, inputs: np.ndarray) -> float:
        """Calculate how grid-like the input distribution is."""
        # Use clustering to detect regular patterns
        try:
            clustering = DBSCAN(eps=0.05, min_samples=3).fit(inputs)
            n_clusters = len(set(clustering.labels_)) - (1 if -1 in clustering.labels_ else 0)
            
            # Many small clusters = grid-like pattern
            if len(inputs) > 0:
                return min(1.0, n_clusters / (len(inputs) * 0.1))
        except:
            pass
        return 0.0
    
    def _hash_input(self, input_vector: np.ndarray) -> str:
        return hashlib.sha256(input_vector.tobytes()).hexdigest()[:16]
    
    def get_user_risk_score(self, user_id: str) -> float:
        """Get overall risk score for a user."""
        user_alerts = [a for a in self.alerts if a.user_id == user_id]
        if not user_alerts:
            return 0.0
        return min(1.0, sum(a.confidence for a in user_alerts) / len(user_alerts))


class OutputProtection:
    """Protect model outputs from extraction."""
    
    @staticmethod
    def add_noise(predictions: np.ndarray, noise_level: float = 0.01) -> np.ndarray:
        """Add small noise to predictions to hinder extraction."""
        noise = np.random.normal(0, noise_level, predictions.shape)
        return predictions + noise
    
    @staticmethod
    def round_confidence(predictions: np.ndarray, precision: int = 2) -> np.ndarray:
        """Round confidence scores to reduce information leakage."""
        return np.round(predictions, precision)
    
    @staticmethod
    def top_k_only(predictions: np.ndarray, k: int = 3) -> np.ndarray:
        """Only return top-k predictions, zero out others."""
        result = np.zeros_like(predictions)
        top_k_idx = np.argsort(predictions)[-k:]
        result[top_k_idx] = predictions[top_k_idx]
        return result
```

```python
import numpy as np
import torch
import torch.nn as nn
from torch.utils.data import DataLoader
from typing import Tuple, Optional
from dataclasses import dataclass
import math

@dataclass
class PrivacyBudget:
    epsilon: float  # Privacy loss
    delta: float    # Failure probability
    queries_used: int
    queries_remaining: int

class DifferentialPrivacyTrainer:
    """Train ML models with differential privacy guarantees."""
    
    def __init__(self, 
                 model: nn.Module,
                 epsilon: float = 1.0,
                 delta: float = 1e-5,
                 max_grad_norm: float = 1.0,
                 noise_multiplier: float = 1.1):
        """
        Args:
            epsilon: Privacy budget (lower = more private)
            delta: Probability of privacy breach
            max_grad_norm: Clip gradients to this norm
            noise_multiplier: Noise added to gradients
        """
        self.model = model
        self.epsilon = epsilon
        self.delta = delta
        self.max_grad_norm = max_grad_norm
        self.noise_multiplier = noise_multiplier
        self.privacy_spent = 0.0
    
    def train_step(self, batch_x: torch.Tensor, batch_y: torch.Tensor,
                   optimizer: torch.optim.Optimizer,
                   loss_fn: nn.Module) -> Tuple[float, float]:
        """
        Perform one DP-SGD training step.
        
        Returns:
            (loss, privacy_spent_this_step)
        """
        self.model.train()
        
        # Compute per-sample gradients
        per_sample_grads = self._compute_per_sample_gradients(
            batch_x, batch_y, loss_fn
        )
        
        # Clip gradients
        clipped_grads = self._clip_gradients(per_sample_grads)
        
        # Add noise
        noisy_grads = self._add_noise(clipped_grads)
        
        # Average gradients
        avg_grads = {k: v.mean(dim=0) for k, v in noisy_grads.items()}
        
        # Apply gradients
        optimizer.zero_grad()
        for name, param in self.model.named_parameters():
            if param.requires_grad:
                param.grad = avg_grads.get(name)
        optimizer.step()
        
        # Track privacy budget
        privacy_cost = self._compute_privacy_cost(len(batch_x))
        self.privacy_spent += privacy_cost
        
        # Compute loss for logging
        with torch.no_grad():
            output = self.model(batch_x)
            loss = loss_fn(output, batch_y).item()
        
        return loss, privacy_cost
    
    def _compute_per_sample_gradients(self, x, y, loss_fn):
        """Compute gradients for each sample individually."""
        per_sample_grads = {}
        
        for i in range(len(x)):
            self.model.zero_grad()
            output = self.model(x[i:i+1])
            loss = loss_fn(output, y[i:i+1])
            loss.backward()
            
            for name, param in self.model.named_parameters():
                if param.requires_grad:
                    if name not in per_sample_grads:
                        per_sample_grads[name] = []
                    per_sample_grads[name].append(param.grad.clone())
        
        # Stack into tensors
        return {k: torch.stack(v) for k, v in per_sample_grads.items()}
    
    def _clip_gradients(self, per_sample_grads):
        """Clip each sample's gradients to max_grad_norm."""
        clipped = {}
        
        for name, grads in per_sample_grads.items():
            # Compute per-sample gradient norms
            norms = grads.view(grads.shape[0], -1).norm(dim=1)
            
            # Clip factor
            clip_factor = self.max_grad_norm / (norms + 1e-6)
            clip_factor = torch.clamp(clip_factor, max=1.0)
            
            # Apply clipping
            clipped[name] = grads * clip_factor.view(-1, *([1] * (grads.dim() - 1)))
        
        return clipped
    
    def _add_noise(self, clipped_grads):
        """Add Gaussian noise for differential privacy."""
        noisy = {}
        noise_std = self.noise_multiplier * self.max_grad_norm
        
        for name, grads in clipped_grads.items():
            noise = torch.normal(0, noise_std, size=grads.shape)
            noisy[name] = grads + noise
        
        return noisy
    
    def _compute_privacy_cost(self, batch_size: int) -> float:
        """Compute privacy cost of this step (simplified)."""
        # Using basic composition (real implementation uses RDP)
        return (self.noise_multiplier ** -2) * (batch_size ** -1)
    
    def get_privacy_budget(self) -> PrivacyBudget:
        """Get current privacy budget status."""
        remaining = max(0, self.epsilon - self.privacy_spent)
        return PrivacyBudget(
            epsilon=self.epsilon,
            delta=self.delta,
            queries_used=int(self.privacy_spent * 1000),
            queries_remaining=int(remaining * 1000)
        )
```

```python
# security_linter.py
import ast
import re
from dataclasses import dataclass
from typing import List

@dataclass
class SecurityFinding:
    file: str
    line: int
    rule: str
    severity: str
    message: str
    code_snippet: str

class SecurityLinter(ast.NodeVisitor):
    """Find security issues in Python code."""
    
    def __init__(self, filename: str, source: str):
        self.filename = filename
        self.source = source
        self.lines = source.split('\n')
        self.findings: List[SecurityFinding] = []
    
    def visit_Call(self, node):
        # Check for dangerous functions
        self._check_dangerous_calls(node)
        self._check_sql_injection(node)
        self._check_command_injection(node)
        self._check_insecure_random(node)
        self.generic_visit(node)
    
    def _check_dangerous_calls(self, node):
        dangerous = {
            'eval': 'Use ast.literal_eval() instead',
            'exec': 'Avoid dynamic code execution',
            'pickle.loads': 'Pickle is unsafe for untrusted data',
            'yaml.load': 'Use yaml.safe_load() instead',
        }
        
        func_name = self._get_func_name(node)
        if func_name in dangerous:
            self._add_finding(
                node.lineno,
                'DANGEROUS_FUNCTION',
                'HIGH',
                f"Dangerous function '{func_name}': {dangerous[func_name]}"
            )
    
    def _check_sql_injection(self, node):
        func_name = self._get_func_name(node)
        if func_name in ('execute', 'executemany'):
            if len(node.args) > 0:
                # Check if query is a formatted string
                if isinstance(node.args[0], ast.BinOp):
                    if isinstance(node.args[0].op, (ast.Mod, ast.Add)):
                        self._add_finding(
                            node.lineno,
                            'SQL_INJECTION',
                            'CRITICAL',
                            'Possible SQL injection: use parameterized queries'
                        )
                elif isinstance(node.args[0], ast.JoinedStr):  # f-string
                    self._add_finding(
                        node.lineno,
                        'SQL_INJECTION',
                        'CRITICAL',
                        'SQL injection via f-string: use parameterized queries'
                    )
    
    def _check_command_injection(self, node):
        func_name = self._get_func_name(node)
        if func_name in ('os.system', 'subprocess.call', 'subprocess.run'):
            # Check for shell=True
            for keyword in node.keywords:
                if keyword.arg == 'shell' and self._is_true(keyword.value):
                    self._add_finding(
                        node.lineno,
                        'COMMAND_INJECTION',
                        'HIGH',
                        'shell=True with untrusted input enables command injection'
                    )
    
    def _check_insecure_random(self, node):
        func_name = self._get_func_name(node)
        if func_name in ('random.random', 'random.randint', 'random.choice'):
            self._add_finding(
                node.lineno,
                'INSECURE_RANDOM',
                'MEDIUM',
                'Use secrets module for security-sensitive randomness'
            )
    
    def _get_func_name(self, node) -> str:
        if isinstance(node.func, ast.Name):
            return node.func.id
        elif isinstance(node.func, ast.Attribute):
            parts = []
            current = node.func
            while isinstance(current, ast.Attribute):
                parts.append(current.attr)
                current = current.value
            if isinstance(current, ast.Name):
                parts.append(current.id)
            return '.'.join(reversed(parts))
        return ''
    
    def _is_true(self, node) -> bool:
        return isinstance(node, ast.Constant) and node.value is True
    
    def _add_finding(self, line: int, rule: str, severity: str, message: str):
        code = self.lines[line - 1] if line <= len(self.lines) else ""
        self.findings.append(SecurityFinding(
            file=self.filename,
            line=line,
            rule=rule,
            severity=severity,
            message=message,
            code_snippet=code.strip()
        ))
    
    def scan(self) -> List[SecurityFinding]:
        tree = ast.parse(self.source)
        self.visit(tree)
        return self.findings

def scan_file(filepath: str) -> List[SecurityFinding]:
    with open(filepath, 'r') as f:
        source = f.read()
    linter = SecurityLinter(filepath, source)
    return linter.scan()
```

```python
        with open(req_file) as f:
            for line in f:
                line = line.strip()
                if not line or line.startswith('#'):
                    continue
                
                # Parse package==version
                if '==' in line:
                    package, version = line.split('==')
                    self._check_osv(package.strip(), version.strip(), 'PyPI')
    
    def _check_osv(self, package: str, version: str, ecosystem: str):
        """Check package against OSV database."""
        payload = {
            "package": {"name": package, "ecosystem": ecosystem},
            "version": version
        }
        
        try:
            response = requests.post(self.OSV_API, json=payload)
            data = response.json()
            
            for vuln in data.get('vulns', []):
                severity = self._get_severity(vuln)
                fixed = self._get_fixed_version(vuln)
                
                self.vulnerabilities.append(Vulnerability(
                    cve_id=vuln.get('id', 'Unknown'),
                    package=package,
                    installed_version=version,
                    fixed_version=fixed,
                    severity=severity,
                    description=vuln.get('summary', '')[:200],
                    cvss_score=self._extract_cvss(vuln)
                ))
        except Exception as e:
            print(f"Error checking {package}: {e}")
    
    def _get_severity(self, vuln: dict) -> str:
        severity_map = {
            (0, 4): 'LOW',
            (4, 7): 'MEDIUM', 
            (7, 9): 'HIGH',
            (9, 10.1): 'CRITICAL'
        }
        cvss = self._extract_cvss(vuln)
        for (low, high), sev in severity_map.items():
            if low <= cvss < high:
                return sev
        return 'UNKNOWN'
    
    def _extract_cvss(self, vuln: dict) -> float:
        for severity in vuln.get('severity', []):
            if 'score' in severity:
                return float(severity['score'])
        return 0.0
    
    def _get_fixed_version(self, vuln: dict) -> Optional[str]:
        for affected in vuln.get('affected', []):
            for range_info in affected.get('ranges', []):
                for event in range_info.get('events', []):
                    if 'fixed' in event:
                        return event['fixed']
        return None
    
    def generate_report(self) -> str:
        """Generate vulnerability report."""
        report = "# Dependency Vulnerability Report\n\n"
        report += f"**Scanned:** {self.project_path}\n"
        report += f"**Vulnerabilities Found:** {len(self.vulnerabilities)}\n\n"
        
        # Group by severity
        by_severity = {'CRITICAL': [], 'HIGH': [], 'MEDIUM': [], 'LOW': []}
        for v in self.vulnerabilities:
            if v.severity in by_severity:
                by_severity[v.severity].append(v)
        
        for severity in ['CRITICAL', 'HIGH', 'MEDIUM', 'LOW']:
            vulns = by_severity[severity]
            if vulns:
                report += f"## {severity} ({len(vulns)})\n\n"
                for v in vulns:
                    report += f"### {v.cve_id}\n"
                    report += f"- **Package:** {v.package} {v.installed_version}\n"
                    report += f"- **Fixed in:** {v.fixed_version or 'No fix available'}\n"
                    report += f"- **CVSS:** {v.cvss_score}\n"
                    report += f"- **Description:** {v.description}\n\n"
        
        return report
```

```python
import json
from dataclasses import dataclass
from typing import List, Dict, Optional
from urllib.parse import urljoin
import yaml

@dataclass
class APIEndpoint:
    path: str
    method: str
    parameters: List[Dict]
    auth_required: bool

@dataclass
class SecurityFinding:
    endpoint: str
    method: str
    vulnerability: str
    severity: str
    owasp_api: str
    evidence: str
    remediation: str

class APISecurityTester:
    """Comprehensive API security testing framework."""
    
    OWASP_API_TOP_10 = {
        'BOLA': 'API1:2023 Broken Object Level Authorization',
        'BROKEN_AUTH': 'API2:2023 Broken Authentication',
        'INJECTION': 'API8:2023 Security Misconfiguration',
        'MASS_ASSIGNMENT': 'API3:2023 Broken Object Property Level Authorization',
        'RATE_LIMIT': 'API4:2023 Unrestricted Resource Consumption',
        'BFLA': 'API5:2023 Broken Function Level Authorization',
    }
    
    def __init__(self, base_url: str, auth_token: Optional[str] = None):
        self.base_url = base_url.rstrip('/')
        self.auth_token = auth_token
        self.endpoints: List[APIEndpoint] = []
        self.findings: List[SecurityFinding] = []
        self.session = requests.Session()
        
        if auth_token:
            self.session.headers['Authorization'] = f'Bearer {auth_token}'
    
    def load_openapi(self, spec_path: str):
        """Load endpoints from OpenAPI spec."""
        with open(spec_path) as f:
            if spec_path.endswith('.yaml') or spec_path.endswith('.yml'):
                spec = yaml.safe_load(f)
            else:
                spec = json.load(f)
        
        for path, methods in spec.get('paths', {}).items():
            for method, details in methods.items():
                if method.upper() in ['GET', 'POST', 'PUT', 'DELETE', 'PATCH']:
                    params = []
                    for param in details.get('parameters', []):
                        params.append({
                            'name': param['name'],
                            'in': param['in'],
                            'required': param.get('required', False)
                        })
                    
                    self.endpoints.append(APIEndpoint(
                        path=path,
                        method=method.upper(),
                        parameters=params,
                        auth_required='security' in details
                    ))
    
    def run_all_tests(self) -> List[SecurityFinding]:
        """Run comprehensive security tests."""
        for endpoint in self.endpoints:
            self._test_bola(endpoint)
            self._test_injection(endpoint)
            self._test_rate_limiting(endpoint)
            self._test_security_headers(endpoint)
            self._test_mass_assignment(endpoint)
        
        return self.findings
    
    def _test_bola(self, endpoint: APIEndpoint):
        """Test for Broken Object Level Authorization."""
        # Look for ID parameters
        id_params = [p for p in endpoint.parameters 
                    if 'id' in p['name'].lower() or p['name'] in ['id', 'userId', 'orderId']]
        
        if not id_params:
            return
        
        # Try accessing other users' resources
        test_ids = ['1', '2', '999', '0', '-1', 'admin']
        
        for param in id_params:
            for test_id in test_ids:
                path = endpoint.path.replace(f"{{{param['name']}}}", test_id)
                url = urljoin(self.base_url, path)
                
                try:
                    resp = self.session.request(endpoint.method, url)
                    
                    if resp.status_code == 200:
                        self.findings.append(SecurityFinding(
                            endpoint=endpoint.path,
                            method=endpoint.method,
                            vulnerability="BOLA/IDOR",
                            severity="HIGH",
                            owasp_api=self.OWASP_API_TOP_10['BOLA'],
                            evidence=f"Accessed {path} with ID {test_id}",
                            remediation="Implement object-level authorization checks"
                        ))
                        break
                except:
                    pass
    
    def _test_injection(self, endpoint: APIEndpoint):
        """Test for injection vulnerabilities."""
        injection_payloads = {
            'sql': ["' OR '1'='1", "1; DROP TABLE users--", "1 UNION SELECT * FROM users"],
            'nosql': ['{"$gt": ""}', '{"$ne": null}'],
            'cmd': ['; ls -la', '| cat /etc/passwd', '`whoami`']
        }
        
        for param in endpoint.parameters:
            for injection_type, payloads in injection_payloads.items():
                for payload in payloads:
                    url = urljoin(self.base_url, endpoint.path)
                    
                    if param['in'] == 'query':
                        params = {param['name']: payload}
                        try:
                            resp = self.session.request(endpoint.method, url, params=params)
                            if self._check_injection_response(resp, injection_type):
                                self.findings.append(SecurityFinding(
                                    endpoint=endpoint.path,
                                    method=endpoint.method,
                                    vulnerability=f"{injection_type.upper()} Injection",
                                    severity="CRITICAL",
                                    owasp_api=self.OWASP_API_TOP_10['INJECTION'],
                                    evidence=f"Payload: {payload}",
                                    remediation="Use parameterized queries, input validation"
                                ))
                                return
                        except:
                            pass
    
    def _test_rate_limiting(self, endpoint: APIEndpoint):
        """Test for missing rate limiting."""
        url = urljoin(self.base_url, endpoint.path)
        
        # Send 50 rapid requests
        success_count = 0
        for _ in range(50):
            try:
                resp = self.session.request(endpoint.method, url)
                if resp.status_code not in [429, 503]:
                    success_count += 1
            except:
                pass
        
        if success_count >= 45:  # 90%+ success = no rate limiting
            self.findings.append(SecurityFinding(
                endpoint=endpoint.path,
                method=endpoint.method,
                vulnerability="Missing Rate Limiting",
                severity="MEDIUM",
                owasp_api=self.OWASP_API_TOP_10['RATE_LIMIT'],
                evidence=f"{success_count}/50 requests succeeded",
                remediation="Implement rate limiting (e.g., 100 req/min)"
            ))
    
    def _test_security_headers(self, endpoint: APIEndpoint):
        """Check for security headers."""
        url = urljoin(self.base_url, endpoint.path)
        required_headers = [
            'X-Content-Type-Options',
            'X-Frame-Options', 
            'Strict-Transport-Security',
            'Cache-Control'
        ]
        
        try:
            resp = self.session.request(endpoint.method, url)
            missing = [h for h in required_headers if h not in resp.headers]
            
            if missing:
                self.findings.append(SecurityFinding(
                    endpoint=endpoint.path,
                    method=endpoint.method,
                    vulnerability="Missing Security Headers",
                    severity="LOW",
                    owasp_api=self.OWASP_API_TOP_10['INJECTION'],
                    evidence=f"Missing: {', '.join(missing)}",
                    remediation="Add security headers to all responses"
                ))
        except:
            pass
    
    def _test_mass_assignment(self, endpoint: APIEndpoint):
        """Test for mass assignment vulnerabilities."""
        if endpoint.method not in ['POST', 'PUT', 'PATCH']:
            return
        
        url = urljoin(self.base_url, endpoint.path)
        
        # Try adding privileged fields
        malicious_fields = {
            'role': 'admin',
            'isAdmin': True,
            'permissions': ['all'],
            'verified': True,
            'balance': 999999
        }
        
        try:
            resp = self.session.request(endpoint.method, url, json=malicious_fields)
            
            if resp.status_code in [200, 201]:
                response_data = resp.json()
                for field in malicious_fields:
                    if field in str(response_data):
                        self.findings.append(SecurityFinding(
                            endpoint=endpoint.path,
                            method=endpoint.method,
                            vulnerability="Mass Assignment",
                            severity="HIGH",
                            owasp_api=self.OWASP_API_TOP_10['MASS_ASSIGNMENT'],
                            evidence=f"Field '{field}' accepted",
                            remediation="Whitelist allowed fields explicitly"
                        ))
                        return
        except:
            pass
    
    def _check_injection_response(self, resp, injection_type: str) -> bool:
        """Check if response indicates successful injection."""
        error_patterns = {
            'sql': ['sql', 'mysql', 'sqlite', 'postgresql', 'syntax error'],
            'nosql': ['mongodb', 'mongoose', 'bson'],
            'cmd': ['bin', 'root:', '/home', 'command not found']
        }
        
        text = resp.text.lower()
        return any(p in text for p in error_patterns.get(injection_type, []))
    
    def generate_report(self) -> str:
        """Generate security testing report."""
        report = "# API Security Test Report\n\n"
        report += f"**Target:** {self.base_url}\n"
        report += f"**Endpoints Tested:** {len(self.endpoints)}\n"
        report += f"**Findings:** {len(self.findings)}\n\n"
        
        by_severity = {'CRITICAL': [], 'HIGH': [], 'MEDIUM': [], 'LOW': []}
        for f in self.findings:
            by_severity[f.severity].append(f)
        
        for severity in ['CRITICAL', 'HIGH', 'MEDIUM', 'LOW']:
            findings = by_severity[severity]
            if findings:
                report += f"## {severity} ({len(findings)})\n\n"
                for f in findings:
                    report += f"### {f.vulnerability}\n"
                    report += f"- **Endpoint:** `{f.method} {f.endpoint}`\n"
                    report += f"- **OWASP:** {f.owasp_api}\n"
                    report += f"- **Evidence:** {f.evidence}\n"
                    report += f"- **Fix:** {f.remediation}\n\n"
        
        return report
```

```python
import re
import os
from dataclasses import dataclass
from typing import List, Pattern
from pathlib import Path

@dataclass
class Secret:
    file: str
    line: int
    secret_type: str
    value: str  # Partially masked
    severity: str

class SecretsScanner:
    """Scan code for exposed secrets and credentials."""
    
    PATTERNS = {
        'AWS Access Key': {
            'pattern': r'AKIA[0-9A-Z]{16}',
            'severity': 'CRITICAL'
        },
        'AWS Secret Key': {
            'pattern': r'(?i)aws(.{0,20})?(?-i)[\'"][0-9a-zA-Z/+]{40}[\'"]',
            'severity': 'CRITICAL'
        },
        'GitHub Token': {
            'pattern': r'ghp_[a-zA-Z0-9]{36}',
            'severity': 'CRITICAL'
        },
        'Generic API Key': {
            'pattern': r'(?i)(api[_-]?key|apikey)[\s]*[=:]\s*[\'"][a-zA-Z0-9]{20,}[\'"]',
            'severity': 'HIGH'
        },
        'Private Key': {
            'pattern': r'-----BEGIN (RSA |EC |DSA )?PRIVATE KEY-----',
            'severity': 'CRITICAL'
        },
        'Password in URL': {
            'pattern': r'(?i)://[^:]+:[^@]+@',
            'severity': 'HIGH'
        },
        'JWT Token': {
            'pattern': r'eyJ[a-zA-Z0-9_-]*\.eyJ[a-zA-Z0-9_-]*\.[a-zA-Z0-9_-]*',
            'severity': 'MEDIUM'
        },
        'Slack Token': {
            'pattern': r'xox[baprs]-[0-9a-zA-Z]{10,48}',
            'severity': 'HIGH'
        },
        'Database URL': {
            'pattern': r'(?i)(postgres|mysql|mongodb)://[^\s]+',
            'severity': 'HIGH'
        }
    }
    
    IGNORE_DIRS = {'.git', 'node_modules', '__pycache__', 'venv', '.env'}
    IGNORE_EXTENSIONS = {'.jpg', '.png', '.gif', '.ico', '.pdf', '.zip'}
    
    def __init__(self, repo_path: str):
        self.repo_path = Path(repo_path)
        self.secrets: List[Secret] = []
    
    def scan(self) -> List[Secret]:
        """Scan repository for secrets."""
        for file_path in self._get_files():
            self._scan_file(file_path)
        return self.secrets
    
    def _get_files(self):
        """Get all scannable files."""
        for root, dirs, files in os.walk(self.repo_path):
            # Skip ignored directories
            dirs[:] = [d for d in dirs if d not in self.IGNORE_DIRS]
            
            for file in files:
                file_path = Path(root) / file
                if file_path.suffix not in self.IGNORE_EXTENSIONS:
                    yield file_path
    
    def _scan_file(self, file_path: Path):
        """Scan a single file for secrets."""
        try:
            with open(file_path, 'r', errors='ignore') as f:
                lines = f.readlines()
        except:
            return
        
        for line_num, line in enumerate(lines, 1):
            for secret_type, config in self.PATTERNS.items():
                matches = re.finditer(config['pattern'], line)
                for match in matches:
                    value = match.group()
                    self.secrets.append(Secret(
                        file=str(file_path.relative_to(self.repo_path)),
                        line=line_num,
                        secret_type=secret_type,
                        value=self._mask_secret(value),
                        severity=config['severity']
                    ))
    
    def _mask_secret(self, value: str) -> str:
        """Mask middle portion of secret."""
        if len(value) <= 10:
            return value[:2] + '*' * (len(value) - 4) + value[-2:]
        return value[:5] + '*' * 10 + value[-5:]
    
    def generate_report(self) -> str:
        """Generate secrets report."""
        report = "# Secrets Scan Report\n\n"
        report += f"**Repository:** {self.repo_path}\n"
        report += f"**Secrets Found:** {len(self.secrets)}\n\n"
        
        if not self.secrets:
            report += "✅ No secrets detected!\n"
            return report
        
        for secret in sorted(self.secrets, key=lambda s: s.severity):
            report += f"## {secret.secret_type}\n"
            report += f"- **File:** {secret.file}:{secret.line}\n"
            report += f"- **Severity:** {secret.severity}\n"
            report += f"- **Value:** `{secret.value}`\n\n"
        
        return report
```

```python
# aws_security_checker.py
import boto3
from dataclasses import dataclass
from typing import List, Optional

@dataclass
class SecurityFinding:
    resource_type: str
    resource_id: str
    check_name: str
    severity: str
    description: str
    remediation: str
    cis_benchmark: Optional[str] = None

class AWSSecurityChecker:
    """Check AWS account for common security misconfigurations."""
    
    def __init__(self):
        self.findings: List[SecurityFinding] = []
        self.iam = boto3.client('iam')
        self.s3 = boto3.client('s3')
        self.ec2 = boto3.client('ec2')
    
    def run_all_checks(self):
        print("Running AWS Security Checks...")
        self._check_root_access_keys()
        self._check_mfa_on_root()
        self._check_public_s3_buckets()
        self._check_security_groups()
        self._check_iam_password_policy()
        self._check_cloudtrail_enabled()
        return self.findings
    
    def _check_root_access_keys(self):
        """CIS 1.4: Ensure no root access keys exist."""
        try:
            summary = self.iam.get_account_summary()
            if summary['SummaryMap']['AccountAccessKeysPresent'] > 0:
                self.findings.append(SecurityFinding(
                    resource_type='IAM',
                    resource_id='root',
                    check_name='Root Access Keys',
                    severity='CRITICAL',
                    description='Root account has access keys',
                    remediation='Delete root access keys immediately',
                    cis_benchmark='CIS 1.4'
                ))
        except Exception as e:
            print(f"Error checking root keys: {e}")
    
    def _check_public_s3_buckets(self):
        """Check for publicly accessible S3 buckets."""
        try:
            buckets = self.s3.list_buckets()['Buckets']
            for bucket in buckets:
                bucket_name = bucket['Name']
                try:
                    # Check bucket ACL
                    acl = self.s3.get_bucket_acl(Bucket=bucket_name)
                    for grant in acl['Grants']:
                        grantee = grant.get('Grantee', {})
                        if grantee.get('URI') in [
                            'http://acs.amazonaws.com/groups/global/AllUsers',
                            'http://acs.amazonaws.com/groups/global/AuthenticatedUsers'
                        ]:
                            self.findings.append(SecurityFinding(
                                resource_type='S3',
                                resource_id=bucket_name,
                                check_name='Public Bucket ACL',
                                severity='HIGH',
                                description=f'Bucket {bucket_name} has public ACL',
                                remediation='Remove public ACL and enable Block Public Access'
                            ))
                except Exception:
                    pass  # Bucket might not have ACL access
        except Exception as e:
            print(f"Error checking S3 buckets: {e}")
    
    def _check_security_groups(self):
        """Check for overly permissive security groups."""
        try:
            sgs = self.ec2.describe_security_groups()['SecurityGroups']
            for sg in sgs:
                for rule in sg.get('IpPermissions', []):
                    for ip_range in rule.get('IpRanges', []):
                        if ip_range.get('CidrIp') == '0.0.0.0/0':
                            port = rule.get('FromPort', 'all')
                            if port in [22, 3389, 'all'] or port == -1:
                                self.findings.append(SecurityFinding(
                                    resource_type='EC2',
                                    resource_id=sg['GroupId'],
                                    check_name='Open Admin Ports',
                                    severity='HIGH',
                                    description=f"SG {sg['GroupId']} allows {port} from 0.0.0.0/0",
                                    remediation='Restrict to specific IPs or use VPN'
                                ))
        except Exception as e:
            print(f"Error checking security groups: {e}")
    
    def generate_report(self) -> str:
        report = "# AWS Security Audit Report\n\n"
        report += f"## Summary\nTotal Findings: {len(self.findings)}\n\n"
        
        by_severity = {}
        for f in self.findings:
            if f.severity not in by_severity:
                by_severity[f.severity] = []
            by_severity[f.severity].append(f)
        
        for severity in ['CRITICAL', 'HIGH', 'MEDIUM', 'LOW']:
            if severity in by_severity:
                report += f"### {severity} ({len(by_severity[severity])})\n"
                for finding in by_severity[severity]:
                    report += f"""
**{finding.check_name}**
- Resource: {finding.resource_type}/{finding.resource_id}
- Issue: {finding.description}
- Fix: {finding.remediation}
{f'- CIS: {finding.cis_benchmark}' if finding.cis_benchmark else ''}
"""
        return report
```

```python
import json
import re
from pathlib import Path
from dataclasses import dataclass
from typing import List, Dict

@dataclass
class IaCFinding:
    file: str
    line: int
    resource: str
    rule: str
    severity: str
    description: str
    remediation: str

class TerraformScanner:
    """Scan Terraform files for security misconfigurations."""
    
    RULES = {
        'S3_PUBLIC_ACCESS': {
            'pattern': r'acl\s*=\s*["\']public',
            'severity': 'CRITICAL',
            'description': 'S3 bucket allows public access',
            'remediation': 'Set acl = "private" and enable Block Public Access'
        },
        'UNENCRYPTED_EBS': {
            'pattern': r'resource\s+"aws_ebs_volume".*?(?!encrypted\s*=\s*true)',
            'severity': 'HIGH',
            'description': 'EBS volume not encrypted',
            'remediation': 'Add encrypted = true'
        },
        'OPEN_SECURITY_GROUP': {
            'pattern': r'cidr_blocks\s*=\s*\[\s*"0\.0\.0\.0/0"\s*\]',
            'severity': 'HIGH',
            'description': 'Security group open to the world',
            'remediation': 'Restrict to specific CIDR ranges'
        },
        'HARDCODED_SECRET': {
            'pattern': r'(password|secret|api_key)\s*=\s*"[^"]{8,}"',
            'severity': 'CRITICAL',
            'description': 'Hardcoded secret in terraform',
            'remediation': 'Use variables and secrets manager'
        },
        'NO_LOGGING': {
            'pattern': r'resource\s+"aws_s3_bucket"(?!.*logging)',
            'severity': 'MEDIUM',
            'description': 'S3 bucket without access logging',
            'remediation': 'Enable access logging'
        }
    }
    
    def __init__(self, path: str):
        self.path = Path(path)
        self.findings: List[IaCFinding] = []
    
    def scan(self) -> List[IaCFinding]:
        """Scan all .tf files."""
        for tf_file in self.path.rglob('*.tf'):
            self._scan_file(tf_file)
        return self.findings
    
    def _scan_file(self, file_path: Path):
        """Scan a single Terraform file."""
        with open(file_path) as f:
            content = f.read()
            lines = content.split('\n')
        
        for rule_name, rule_config in self.RULES.items():
            for match in re.finditer(rule_config['pattern'], content, re.IGNORECASE | re.DOTALL):
                # Find line number
                line_num = content[:match.start()].count('\n') + 1
                
                self.findings.append(IaCFinding(
                    file=str(file_path),
                    line=line_num,
                    resource=self._extract_resource_name(content, match.start()),
                    rule=rule_name,
                    severity=rule_config['severity'],
                    description=rule_config['description'],
                    remediation=rule_config['remediation']
                ))
    
    def _extract_resource_name(self, content: str, position: int) -> str:
        """Extract the resource name for context."""
        # Look backwards for resource declaration
        before = content[:position]
        match = re.search(r'resource\s+"([^"]+)"\s+"([^"]+)"', before[::-1])
        if match:
            return f"{match.group(1)[::-1]}.{match.group(2)[::-1]}"
        return "unknown"
```

### PROJECT 16: Cloud IAM Analyzer
**Difficulty:** Medium-Hard | **Time:** 20-25 hours | **Impact:** HIGH

#### What You'll Build
Analyze IAM policies for excessive permissions and security risks.

#### Core Implementation

`iam_analyzer.py`:
```python
import boto3
import json
from dataclasses import dataclass
from typing import List, Dict, Set

@dataclass
class IAMRisk:
    principal: str
    principal_type: str
    risk: str
    severity: str
    permissions: List[str]
    recommendation: str

class IAMAnalyzer:
    """Analyze AWS IAM for security risks."""
    
    DANGEROUS_ACTIONS = {
        'iam:*': 'CRITICAL',
        'iam:CreateUser': 'HIGH',
        'iam:CreateAccessKey': 'HIGH',
        'iam:AttachUserPolicy': 'HIGH',
        'iam:PutUserPolicy': 'HIGH',
        's3:*': 'HIGH',
        'ec2:*': 'HIGH',
        'lambda:*': 'MEDIUM',
        'sts:AssumeRole': 'MEDIUM',
        '*:*': 'CRITICAL',
    }
    
    def __init__(self):
        self.iam = boto3.client('iam')
        self.risks: List[IAMRisk] = []
    
    def analyze(self) -> List[IAMRisk]:
        """Run full IAM analysis."""
        self._analyze_users()
        self._analyze_roles()
        self._analyze_policies()
        return self.risks
    
    def _analyze_users(self):
        """Check IAM users for risks."""
        users = self.iam.list_users()['Users']
        
        for user in users:
            username = user['UserName']
            
            # Check for MFA
            mfa = self.iam.list_mfa_devices(UserName=username)
            if not mfa['MFADevices']:
                self.risks.append(IAMRisk(
                    principal=username,
                    principal_type='User',
                    risk='No MFA enabled',
                    severity='HIGH',
                    permissions=[],
                    recommendation='Enable MFA for all IAM users'
                ))
            
            # Check for old access keys
            keys = self.iam.list_access_keys(UserName=username)
            for key in keys['AccessKeyMetadata']:
                age_days = (datetime.utcnow() - key['CreateDate'].replace(tzinfo=None)).days
                if age_days > 90:
                    self.risks.append(IAMRisk(
                        principal=username,
                        principal_type='User',
                        risk=f'Access key older than 90 days ({age_days} days)',
                        severity='MEDIUM',
                        permissions=[],
                        recommendation='Rotate access keys regularly'
                    ))
            
            # Check attached policies
            self._check_user_permissions(username)
    
    def _check_user_permissions(self, username: str):
        """Check user's effective permissions."""
        attached = self.iam.list_attached_user_policies(UserName=username)
        
        for policy in attached['AttachedPolicies']:
            policy_doc = self._get_policy_document(policy['PolicyArn'])
            dangerous = self._find_dangerous_permissions(policy_doc)
            
            if dangerous:
                self.risks.append(IAMRisk(
                    principal=username,
                    principal_type='User',
                    risk='Overly permissive IAM policy',
                    severity=max(self.DANGEROUS_ACTIONS.get(a, 'LOW') for a in dangerous),
                    permissions=list(dangerous),
                    recommendation='Apply least privilege principle'
                ))
    
    def _find_dangerous_permissions(self, policy_doc: dict) -> Set[str]:
        """Find dangerous actions in policy."""
        dangerous = set()
        
        for statement in policy_doc.get('Statement', []):
            if statement.get('Effect') != 'Allow':
                continue
            
            actions = statement.get('Action', [])
            if isinstance(actions, str):
                actions = [actions]
            
            for action in actions:
                if action in self.DANGEROUS_ACTIONS:
                    dangerous.add(action)
                elif action == '*':
                    dangerous.add('*:*')
        
        return dangerous
    
    def _get_policy_document(self, policy_arn: str) -> dict:
        """Get policy document."""
        policy = self.iam.get_policy(PolicyArn=policy_arn)
        version_id = policy['Policy']['DefaultVersionId']
        version = self.iam.get_policy_version(
            PolicyArn=policy_arn,
            VersionId=version_id
        )
        return version['PolicyVersion']['Document']
```
